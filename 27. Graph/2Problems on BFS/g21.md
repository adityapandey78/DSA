class Solution {
private:
    // DFS function for topological sort
    void dfs(int node, vector<int>& vis, stack<int>& st, vector<vector<int>>& adj) {
        vis[node] = 1;

        // Visit all neighbors
        for (auto neighbor : adj[node]) {
            if (!vis[neighbor]) {
                dfs(neighbor, vis, st, adj);
            }
        }

        // Push to stack after visiting all descendants (postorder)
        st.push(node);
    }

public:
    vector<int> topoSort(int V, vector<vector<int>>& edges) {
        // Step 1: Build adjacency list
        vector<vector<int>> adj(V);
        for (auto& edge : edges) {
            int u = edge[0], v = edge[1];
            adj[u].push_back(v);  // u â†’ v
        }

        // Step 2: Perform DFS-based topological sort
        vector<int> vis(V, 0);
        stack<int> st;

        for (int i = 0; i < V; ++i) {
            if (!vis[i]) {
                dfs(i, vis, st, adj);
            }
        }

        // Step 3: Pop from stack to get topological order
        vector<int> topo;
        while (!st.empty()) {
            topo.push_back(st.top());
            st.pop();
        }

        return topo;
    }
};
